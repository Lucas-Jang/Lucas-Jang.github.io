#inlcude<stdio.h>	// 표준 입출력 헤더를 include한다.

int main(void)		//main 시작
{
	int month;	//달을 입력받을 integer 변수 선언
	
	printf("달을 입력하세요 : ");						
	scanf("%d", &month);			//int 변수 month에 scanf로 입력받은 값 저장
	
	switch(month){
		case 12:
		case 1:
		case 2:
			printf("%d월은 겨울입니다.\n",month);	// 변수 month에 12또는 1또는 2가 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 3:
		case 4:
		case 5:
			printf("%d월은 봄입니다.\n",month);	// 변수 month에 3또는 4또는 5가 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 6:
		case 7:
		case 8:
			printf("%d월은 가을입니다.\n",month);	// 변수 month에 6또는 7또는 8이 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 9:
		case 10:
		case 11:
			printf("%d월은 가을입니다.\n",month); //변수 month에 9또는 10 또는 11이 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		default:		//그 이외의 값이 저장되어 있을 경우
			printf("잘못 입력하셨습니다.\n");	//오류임을 출력(deafualt 조건을 항상 처리해주는 것이 좋음)
	}
	return 0;	//0을 return 하고 main 종료
}

#include<stdio.h>	//표준입출력헤더 include

int main(void)		//main 시작
{
	int toss;		//integer 변수 toss 선언
	
	printf("1부터 3사이의 숫자를 입력하세요 : ");	// instruction
	scanf("%d", &toss);				//int 변수 toss에 입력받은 값 저장
	
	switch(toss){					//int 변수 toss에 대한 조건문
		case 1:						//toss에 저장된 값이 1일 경우
			printf("가위~!!\n");		//가위 출력
			break;					// switch 종료
		case 2:						//2일 경우
			printf("바위~!!\n");		//바위 출력
			break;					//그리고 switch문 종료
		case 3:						//3일 경우
			printf("보~!!\n");		//보 출력
			break;					//그리고 switch문 종료
		default:					//1,2,3 이외의 값이 toss에 저장되었을 경우
			printf("1부터 3사이의 숫자가 아닙니다~!!\n");	//예외 발생 출력
	}
	return 0;	//0을 반환하면서 main 종료
}
