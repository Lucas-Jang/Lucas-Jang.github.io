#inlcude<stdio.h>	// 표준 입출력 헤더를 include한다.

int main(void)		//main 시작
{
	int month;	//달을 입력받을 integer 변수 선언
	
	printf("달을 입력하세요 : ");						
	scanf("%d", &month);			//int 변수 month에 scanf로 입력받은 값 저장
	
	switch(month){
		case 12:
		case 1:
		case 2:
			printf("%d월은 겨울입니다.\n",month);	// 변수 month에 12또는 1또는 2가 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 3:
		case 4:
		case 5:
			printf("%d월은 봄입니다.\n",month);	// 변수 month에 3또는 4또는 5가 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 6:
		case 7:
		case 8:
			printf("%d월은 가을입니다.\n",month);	// 변수 month에 6또는 7또는 8이 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		case 9:
		case 10:
		case 11:
			printf("%d월은 가을입니다.\n",month); //변수 month에 9또는 10 또는 11이 저장되어 있을 경우 몇월인지 출력
			break;		//그리고 switch문을 종료
		default:		//그 이외의 값이 저장되어 있을 경우
			printf("잘못 입력하셨습니다.\n");	//오류임을 출력(deafualt 조건을 항상 처리해주는 것이 좋음)
	}
	return 0;	//0을 return 하고 main 종료
}

#include<stdio.h>	//표준입출력헤더 include

int main(void)		//main 시작
{
	int toss;		//integer 변수 toss 선언
	
	printf("1부터 3사이의 숫자를 입력하세요 : ");	// instruction
	scanf("%d", &toss);				//int 변수 toss에 입력받은 값 저장
	
	switch(toss){					//int 변수 toss에 대한 조건문
		case 1:						//toss에 저장된 값이 1일 경우
			printf("가위~!!\n");		//가위 출력
			break;					// switch 종료
		case 2:						//2일 경우
			printf("바위~!!\n");		//바위 출력
			break;					//그리고 switch문 종료
		case 3:						//3일 경우
			printf("보~!!\n");		//보 출력
			break;					//그리고 switch문 종료
		default:					//1,2,3 이외의 값이 toss에 저장되었을 경우
			printf("1부터 3사이의 숫자가 아닙니다~!!\n");	//예외 발생 출력
	}
	return 0;	//0을 반환하면서 main 종료
}

// Example6
#include<stdio.h>		//표준 입출력 헤더 include

int main(void)		//main함수의 시작
{
	int year =0, month = 0, maxDay =30, validinput = 0;	
	//년을 입력받는 year, 월을 입력받는 month, 각 달의 말일을 설정하는 maxDay, 입력이 잘못되었는지 체크하는 validinput 을 정수 자료형으로 선언
	printf("년과 월을 다음과 같이 입력하면 그 달을 말일을 알려드립니다.\n");
	scanf("%d %d", &year, &month);
	//integer 변수 year과 month에 scanf로 입력받은 값을 차례로 저장
	
	switch(month){	//month 변수에 대한 switch 조건문
		case 1:
		case 3:
		case 5:
		case 7:
		case 8:
		case 10:
		case 12:		//1,3,5,7,10,12가 month에 저장되어 있을 경우
			maxDay =31;	//말일을 31로 저장
			validinput = 1;	//유효한 값이 month에 저장되었으므로 1 저장
			break;	//switch문 종료
		case 4:
		case 6:
		case 9:
		case 11:		//4,6,9,11이 month에 저장되어 있을 경우
			maxDay =30;	//말일을 30으로 저장
			validinput = 1;	//유효한 값이 month에 저장되었으므로 1 저장
			break;		//switch문 종료
		case 2:	//2월일 경우는 몇년도인지까지 고려해주어야 함
			if((year%4 == 0) && (year%100 !=0) || (year%400==0))		// 입력받은 year값이 4로 나누어 떨어짐과 동시에 (100으로 나누어 떨어지지 않거나 400으로 나누어떨어질 경우) ==> 윤년조건
				maxDay = 29;	//말일을 29일로 저장
			else
				maxDay =28;		//아니면 28로 저장
			validinput=1;	//역시 유효한 값이 입력되었으므로 1을 저장
			break;	//switch문 종료
		default:	//그 외의 값들이 month에 저장되었을 경우
			validinput =0;	//유효한 값이 저장된 것이 아니므로 0 저장
	}
	if(validinput) 		//조건문 if의 괄호 안에서 0은 FALSE, 그 이외의 자연수는 TRUE
		printf("%d년 %d월의 말일은 %d일입니다.\n",year,month,maxDay);	//형식에 맞게 출력
	else
		printf("입력이 잘못되었습니다.\n");	//month에 유효한 값이 입력되지 않았으므로 에러를 띄운다.
	return 0; 		// 0을 반환하면서 main함수 종료
}
